# 创建


```DB
create table t_book(
    id int,
    name varchar(30),
    author varchar(10),
    price double
)
```

# 增加

`insert into t_book(id,name,author,price) values(1,"いいね本","マリィ",10);`

# 删除

`delete from t_book where id = 2;`

# 查询

```DB
select * from t_book
```

# 修改

```DB
update t_book set price = 36.6
    where id = 1;
```
